## Quick Commands Reference

```bash
# Generate this week's metrics
portfolioWeekly

# Check all project status
standup

# Export evidence
cd ~/Code/rhea
statsPortfolio
axeLocal
todos > ~/portfolio-evidence/rhea-todos-$(date +%Y-%m-%d).txt

# Git activity last 7 days
for dir in ~/Code/rhea ~/Code/iris ~/Code/things-we-do ~/Code/beacons; do
  cd $dir
  echo "=== $(basename $dir) ==="
  git log --since="7 days ago" --pretty=format:"%h - %s"
  echo -e "\n"
done
```
